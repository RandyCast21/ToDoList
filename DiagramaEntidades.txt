Table users {
  id uuid [pk]
  username varchar
  email varchar [unique]
  password_hash varchar
  is_active boolean
  created_at timestamp
  updated_at timestamp
}

Table tasks {
  id uuid [pk]

  owner_id uuid [not null, ref: > users.id]

  parent_id uuid [ref: > tasks.id] // self-relation (árbol)

  title varchar
  description text

  priority varchar // low, medium, high
  difficulty int // 1-5
  status varchar // todo, in_progress, done

  depth int
  order_index int

  created_at timestamp
  updated_at timestamp
}

Table tags {
  id uuid [pk]

  owner_id uuid [not null, ref: > users.id]

  name varchar

  created_at timestamp

  Indexes {
    (owner_id, name) [unique] // evita duplicados por usuario
  }
}

Table task_tags {
  task_id uuid [not null, ref: > tasks.id]
  tag_id uuid [not null, ref: > tags.id]

  Indexes {
    (task_id, tag_id) [unique]
  }
}
