Table users {
  id uuid [pk]

  username varchar
  email varchar [unique]

  password_hash varchar [null] 

  is_active boolean

  created_at timestamp
  updated_at timestamp
}

Table oauth_accounts {
  id uuid [pk]

  user_id uuid [not null, ref: > users.id]

  provider varchar // google, github, apple
  provider_id varchar

  created_at timestamp

  Indexes {
    (provider, provider_id) [unique]
  }
}

Table tasks {
  id uuid [pk]

  owner_id uuid [not null, ref: > users.id]

  parent_id uuid [ref: > tasks.id]

  title varchar
  description text

  priority varchar
  difficulty int
  status varchar

  depth int
  order_index int

  created_at timestamp
  updated_at timestamp
}

Table tags {
  id uuid [pk]

  owner_id uuid [not null, ref: > users.id]

  name varchar

  created_at timestamp

  Indexes {
    (owner_id, name) [unique]
  }
}

Table task_tags {
  task_id uuid [not null, ref: > tasks.id]
  tag_id uuid [not null, ref: > tags.id]

  Indexes {
    (task_id, tag_id) [unique]
  }
}
